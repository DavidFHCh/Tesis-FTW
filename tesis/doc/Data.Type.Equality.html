Class TestEquality {k1} (f : k1 → Type) :=
  { testEquality {a}{b} : f a → f b → option (a :~: b) }.

Instance TestEquality_EqTy {k}{a:k} : `{ TestEquality (fun x ⇒ a :~: x) } :=
  { testEquality := fun a b x y ⇒ match x, y with eq_refl, eq_refl ⇒ Some eq_refl end }.
