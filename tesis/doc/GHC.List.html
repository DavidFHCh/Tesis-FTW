       | cons y ys ⇒ drop (n - 1) ys
       end.

 | nil ⇒ (cons q0 nil)
 | cons y ys ⇒ match (scanr f q0 ys) with
               | cons q qs ⇒ cons (f y q) (cons q qs)
               | nil ⇒ nil
               end
end.
